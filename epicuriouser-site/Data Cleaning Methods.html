def get_normalized_rating(rating):
    if rating == 0:
        return 0
    elif rating == 1.250:
        return 1
    elif rating == 1.875:

    elif rating == 2.500:
        return 2
    elif rating == 3.125:
        return 2.5    
    elif rating == 3.750:
        return 3.0
    elif rating == 4.375:
        return 3.5    
    else:
        return 4
